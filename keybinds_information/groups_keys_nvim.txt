# VIM 
"space"nh --> no highlights

# Window manipulation
"space"wv --> split vertically
"space"wh --> split horizontally
"space"h  --> move left
"space"l  --> move right
"space"k  --> move up
"space"j  --> move down
"space"xw --> close window

# Tab maniuplation
"space"t  --> new tab
"space"i  --> move left
"space"o  --> move right
"space"xt --> close tab

# Special windows
"space"e  --> nvimtreetoggle/explorer
"space"ff --> find files
"space"fg --> live grep
"space"fs --> find strings
"space"fb --> find buffer
"space"ft --> find tags
